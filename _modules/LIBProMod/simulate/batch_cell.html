    """Simulates the model for one batch of pre-calculated number of cells"""

    def __init__(self, cell_chemistry, cell_type, para):
        self.initial = InitialCell(cell_chemistry, cell_type)
        # %%
        self.anode_before_mixing = Anode(cell_chemistry, cell_type,
                                         self.initial.ScrapPercentage * 0, 0, [0, 0, 0, 0])
        self.cathode_before_mixing = Cathode(cell_chemistry, cell_type,
                                             self.initial.ScrapPercentage * 0, 0, [0, 0, 0, 0])
        # Mixing
        self.mixerA = Mixing(self.anode_before_mixing, para)
        self.mixerC = Mixing(self.cathode_before_mixing, para)
        self.anode_after_mixing = Anode(cell_chemistry, cell_type, self.initial.ScrapPercentage * 2 * .3, 1,
                                        self.anode_before_mixing.MassInForNext)
        self.cathode_after_mixing = Cathode(cell_chemistry, cell_type, self.initial.ScrapPercentage * 2 * .3, 1,
                                            self.cathode_before_mixing.MassInForNext)
        self.energy_mixing_anode = self.mixerA.Energy
        self.energy_mixing_cathode = self.mixerC.Energy
        # Drying
        self.cc_anode = CurrentCollector(self.initial.AnodeCCDensity, self.initial.AnodeCC, self.initial.CellWidth,
                                         self.initial.CellLength, self.initial.CellsPerBatch)
        self.cc_cathode = CurrentCollector(self.initial.CathodeCCDensity, self.initial.CathodeCC,
                                           self.initial.CellWidth, self.initial.CellLength, self.initial.CellsPerBatch)
        self.dryerA = Drying(self.anode_after_mixing, self.cc_anode, para.anode_dryer_temperature, para.anode_dryer_time)
        self.dryerC = Drying(self.cathode_after_mixing, self.cc_cathode, para.cathode_dryer_temperature,
                             para.cathode_dryer_time)
        self.anode_after_drying = Anode(cell_chemistry, cell_type,
                                        self.initial.ScrapPercentage * 2 * .1, 1, self.anode_after_mixing.MassInForNext)
        self.cathode_after_drying = Cathode(cell_chemistry, cell_type, self.initial.ScrapPercentage * 2 * .1,
                                            1, self.cathode_after_mixing.MassInForNext)
        self.energy_drying_anode = self.dryerA.get_drying_energy()
        self.energy_drying_cathode = self.dryerC.get_drying_energy()
        # dry room
        self.separator = Separator(self.initial.Separator, self.initial.SeparatorDensity, self.initial.CellLength,
                                   self.initial.CellWidth, self.initial.CellsPerBatch)
        self.electrolyte = Electrolyte(self.initial.ElectrolyteVolume, self.initial.ElectrolyteDensity,
                                       self.initial.CellsPerBatch)
        self.dry_room = DryRoomSs(self.anode_after_drying, self.cathode_after_drying, self.cc_anode, self.cc_cathode,
                                  self.separator, self.electrolyte)
        self.energy_dry_room = self.dry_room.energy(para)
        self.anode_after_dry_room = Anode(cell_chemistry, cell_type, self.initial.ScrapPercentage * 2 * .0725, 1,
                                          self.anode_after_drying.MassInForNext)
        self.cathode_after_dry_room = Cathode(cell_chemistry, cell_type, self.initial.ScrapPercentage * 2 * .0725,
                                              1, self.cathode_after_drying.MassInForNext)
        self.cell_after_dry_room = CellAssembly(cell_chemistry, cell_type, self.anode_after_dry_room,
                                                self.cathode_after_dry_room)
        self.number_of_cells = self.cell_after_dry_room.SolidMass / self.initial.SolidMass1cell
        self.energy_in_batch = self.cell_after_dry_room.TotalMass * self.initial.SpecificEnergy  # Wh
        self.cell_base_material = BaseMaterials(self.cell_after_dry_room, cell_chemistry, cell_type)
        #  other processes
        # coating
        self.coating_energy = (625 * 0.2) * para.time_coating  # J 625 kW/h durr megtec procoater of 40m dryer
        # calendering
        self.calendering_energy = 18 * para.time_calendering  # kW*s TMAX-RM600 https://www.tmaxcn.com/hot-roller-press-calendering-machine-for-li-ion-battery-production-line_p1600.html
        # slitting
        self.slitting_energy = 4 * para.time_slitting  # kW*s 50m/min TOB-110BH-600 https://www.tobmachine.com/500mm-automatic-slitting-machine-for-battery-electrode_p221.html
        # 80m/min 2.2kW TOB-SL650 https://tobmachine.en.alibaba.com/product/60649463616-803657396/Automatic_Slitting_Machine_For_Li_Ion_Battery.html
        # stacking
        self.stacking_energy = 4 * para.time_stacking  # kW*s TOB-S-DP-150 1.3-1.5 s/pcs https://www.tobmachine.com/semi-automatic-battery-stacking-machine-for-battery-electrode-z-fold-assemble_p555.html
        #  electrolyte filling
        self.filling_energy = 10 * para.time_filling  # kW*s TOB-AFMY-200 https://www.tobmachine.com/automatic-battery-filling-system-for-polymer-battery_p236.html
        # forming and grading
        self.formation_energy = 9.5 * para.time_formation  # kW*s  not matching TOB-512-2LP https://www.tobmachine.com/512-channel-5v2a-battery-capacity-tester-for-polymer-battery_p925.html
        # floor heating
        self.floor_area = 30000  # m2
        self.floor_heating_energy = self.floor_area * (
                para.total_time - para.time_formation) / 3600000 * 15 * 10.764  # 7-15 W per feet2(*10.764 perm2)
        # %%  results conversion
        self.energy_kJ = np.array([self.energy_mixing_anode, self.energy_mixing_cathode, 2 * self.coating_energy,
                                   self.energy_drying_anode, self.energy_drying_cathode, self.dryerA.EnergyAir,
                                   self.dryerC.EnergyAir, 2 * self.calendering_energy, 2 * self.slitting_energy,
                                   self.energy_dry_room, self.stacking_energy, self.filling_energy,
                                   self.formation_energy, self.floor_heating_energy])
        self.energy_kJ_order = ('mixing anode', 'mixing cathode', 'coating', 'drying anode', 'drying cathode',
                                'dryer anode air heating', 'dryer cathode air heating', 'calendering', 'slitting',
                                'dry room', 'winding', 'filling', 'formation', 'floor heating')
        self.energy_kJ_per_kg = self.energy_kJ / self.cell_after_dry_room.TotalMass
        self.energy_kJ_per_kWh = self.energy_kJ / (self.cell_after_dry_room.TotalMass * self.initial.SpecificEnergy
                                                   / 1000)
        # self.energy_kJ_per_cell = self.energy_kJ / self.initial.CellsPerBatch
        self.energy_kWh = self.energy_kJ / 3600
        self.energy_kWh_per_kg = self.energy_kJ_per_kg / 3600
        # self.energy_kWh_per_kWh = self.energy_kJ_per_kWh / 3600
        # self.energy_kWh_per_cell = self.energy_kJ_per_cell / 3600
        self.total_energy_kJ = sum(self.energy_kJ)
        self.total_energy_kJ_per_kg = sum(self.energy_kJ_per_kg)
        self.energy_kJ_percent = self.energy_kJ * 100 / self.total_energy_kJ

        #  sets the average values of Dai, so drying and dry room are close to 47-47%
        #  add missing utilities for drying and dry room by multiplying 1.3 and then x everything by efficiency ex:.8
        self.tune1 = 1.38  # 1.766
        self.tune2 = 0.70  # 0.896
        #  or fix the formation to 1.11 of Dai 2019 by dividing 0.896 and then fix drying and dry room ex: 1.766
        self.energy_kJ_tuned = np.array([self.energy_mixing_anode, self.energy_mixing_cathode, 2 * self.coating_energy,
                                         self.energy_drying_anode * self.tune1, self.energy_drying_cathode * self.tune1,
                                         self.dryerA.EnergyAir * self.tune1, self.dryerC.EnergyAir * self.tune1,
                                         2 * self.calendering_energy, 2 * self.slitting_energy,
                                         self.energy_dry_room * self.tune1,
                                         self.stacking_energy, self.filling_energy, self.formation_energy,
                                         self.floor_heating_energy]) / self.tune2
        self.energy_kWh_per_kWh_tuned = self.energy_kJ_tuned / (self.cell_after_dry_room.TotalMass *
                                                                self.initial.SpecificEnergy * 3.6)
        self.energy_kWh_to_export = np.array([self.energy_mixing_anode + self.energy_mixing_cathode,
                                              2 * self.coating_energy, (self.energy_drying_anode +
                                                                        self.dryerA.EnergyAir) * self.tune1,
                                              (self.energy_drying_cathode + self.dryerC.EnergyAir) * self.tune1,
                                              2 * self.calendering_energy, 2 * self.slitting_energy,
                                              self.stacking_energy, self.filling_energy, self.formation_energy,
                                              self.floor_heating_energy, self.energy_dry_room * self.tune1]) / \
                                    self.tune2
        #  optional: sets lower or higher margin of Dai...
        self.tune3 = 1.11 / (self.formation_energy / (self.cell_after_dry_room.TotalMass *
                                                      self.initial.SpecificEnergy * 3.6))
        self.energy_kJ_tuned_option2 = self.tune3 * self.energy_kJ
        self.energy_kWh_per_kWh = self.energy_kJ / (self.cell_after_dry_room.TotalMass *
                                                    self.initial.SpecificEnergy * 3.6)
        self.tunes = tune_option2(self.energy_kWh_per_kWh[9], sum(self.energy_kWh_per_kWh[3:7]), 1)
        self.energy_kJ_tuned_option2[9] = self.tunes[0] * self.energy_dry_room
        self.energy_kJ_tuned_option2[3:7] = self.tunes[1] * self.energy_kJ_tuned_option2[3:7]
        self.energy_kWh_per_kWh_tuned_option2 = self.energy_kJ_tuned_option2 / (self.cell_after_dry_room.TotalMass *
                                                                                self.initial.SpecificEnergy * 3.6)
#######
