[docs]class ActiveMaterial:
    """ Active material mass flow details for Cathode and Anode: through a process step """

    def __init__(self, solid_mass, mass_fraction, step_position, mass_in):
        """
            calculates active material mass flow of a process step

        Args:
            solid_mass: amount of solid mass
            mass_fraction: mass fraction of the active material composition
            step_position: 0 if inlet to the first process step (mixing), any other number for other steps
            mass_in: inlet mass flow to the process step
        """
        if step_position == 0:
            self.ActiveMaterialMass = solid_mass * mass_fraction
        else:
            self.ActiveMaterialMass = mass_in
