[docs]class Separator:
    """Details of separator"""

    def __init__(self, thickness, density, length, width, no_of_cells):
        """
        Calculates volume and mass of separators in a cell.

        Args:
            thickness:
            density:
            length:
            width:
            no_of_cells:
        """
        self.Thickness = thickness
        self.Density = density
        self.Volume = thickness * length * width * no_of_cells
        self.Mass = self.Density * self.Volume

[docs]class CellContainer:
    """Details of cell container"""

    def __init__(self, density, thickness, width, length, wall_thickness, no_of_cells):
        """
        Calculates volume and mass of the container of a cell.

        Args:
            density:
            thickness:
            width:
            length:
            wall_thickness:
            no_of_cells:
        """
        self.Volume = (2 * width + 3 * thickness) * length * wall_thickness * no_of_cells
        self.Mass = density * self.Volume
