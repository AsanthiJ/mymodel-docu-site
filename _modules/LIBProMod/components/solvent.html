from math import exp



[docs]class Solvent:
    """ solvent properties and details"""

    def __init__(self, mass_fraction, total_mass, solvent_type, step_position, mass_in):
        self.MassFraction = mass_fraction
        self.SolventType = solvent_type
        if step_position == 0:
            self.Mass = total_mass * mass_fraction
        else:
            self.Mass = mass_in

        if self.SolventType == 0:  # 0 for water
            self.Density = 1000
            self.MolarMass = 18.01528
            self.Lambda = 2260
            self.HeatCapacity = 4182
        elif self.SolventType == 1:  # 1 for NMP
            self.Density = 1030
            self.MolarMass = 99.133
            self.Lambda = 49.5*1000/self.MolarMass
            self.HeatCapacity = 1900
        else:
            print("Undefined solvent type")
        self.DensityUnits = 'kg/m^3'
        self.MolarMassUnits = 'kg/kmol'
        self.VaporPressureUnits = 'Pa'
        self.LambdaUnits = 'kJ/kg'
        self.HeatCapacityUnits = 'J/kgK'

    def volume(self):
        return self.Mass / self.Density

    def vapor_pressure(self, temperature):
        if self.SolventType == 0:  # 0 for water
            return 133.322 * exp(20.386 - 5132 / temperature)
        elif self.SolventType == 1:  # 1 for NMP
            return 1000 * 1016 * exp(-((temperature - 666.7) / 125.2) ** 2)  # curve fitted from data from ref:
        else:
            print("Undefined solvent type")
