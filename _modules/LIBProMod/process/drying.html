[docs]class Drying:
    """calculates the energy for Drying"""
    VolumeFlowRate = 15.  # m3/s
    Density = 1.24  # kg/m3 air rho at 15C
    MassFlowRate = VolumeFlowRate * Density  # kg/s
    # Height = 0.3  # m total height dryer
    # Length = 80.
    # Velocity = 4. # 12.  # m/s air velocity
    # Width = 1.4  # m dryer width
    # Chambers = 18.

    def __init__(self, electrode, current_collector, temperature, time):
        self.EAD = electrode
        self.CurrentCollector = current_collector
        self.Energy = electrode.WetMassOut[0] * electrode.sv.Lambda  # kJ
        #  self.TotalMass = electrode.sv.Mass + electrode.SolidMass + self.CurrentCollector.Mass
        self.TotalMass = electrode.TotalMass + self.CurrentCollector.Mass
        self.EnergyAir = 1.009 * (temperature - 15) * self.MassFlowRate * time  # cp*dT*m_dot*dt   m_dot=L*u

    def get_drying_energy(self):
        return self.Energy

    def total_mass(self):
        return self.TotalMass
