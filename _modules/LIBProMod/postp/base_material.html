[docs]class BaseMaterials:
    """Converts the cell material to base materials-- per cell calculations"""

    def __init__(self, cell, battery_chemistry, battery_type):
        self.CathodeActiveMaterial = cell.CathodeActiveMaterialMassOut
        if battery_chemistry == 10:
            self.SiliconNanoWire = cell.AnodeActiveMaterialMassOut
            self.Graphite = [0, 0]
            self.BinderPVDF = [0, 0]  # polyvinylidene fluoride
            self.BinderSBRCMC = cell.CathodeBinderMassOut + cell.AnodeBinderMassOut
            self.BinderPVDFAnode = [0, 0]
            self.BinderPVDFCathode = [0, 0]
            self.BinderSBRCMCAnode = cell.AnodeBinderMassOut
            self.BinderSBRCMCCathode = cell.CathodeBinderMassOut
        else:
            self.SiliconNanoWire = [0, 0]
            self.Graphite = cell.AnodeActiveMaterialMassOut
            self.BinderPVDF = 0.6 * (cell.CathodeBinderMassOut + cell.AnodeBinderMassOut)  # polyvinylidene fluoride
            self.BinderSBRCMC = 0.4 * (cell.CathodeBinderMassOut + cell.AnodeBinderMassOut)
            self.BinderPVDFAnode = cell.AnodeBinderMassOut * 0.6
            self.BinderPVDFCathode = cell.CathodeBinderMassOut * 0.6
            self.BinderSBRCMCAnode = cell.AnodeBinderMassOut * 0.4
            self.BinderSBRCMCCathode = cell.CathodeBinderMassOut * 0.4
        self.CarbonBlackAnode = cell.AnodeCarbonMassOut
        self.CarbonBlackCathode = cell.CathodeCarbonMassOut
        self.CarbonBlack = self.CarbonBlackAnode + self.CarbonBlackCathode
        self.CopperFoil = cell.AnodeCCMassOut
        self.CopperTab = [cell.NegativeTerminalMassOut, 0]
        self.Copper = self.CopperFoil + self.CopperTab
        self.AluminumFoil = cell.CathodeCCMassOut
        self.AluminumTab = [cell.PositiveTerminalMassOut, 0]
        self.Aluminum = self.AluminumFoil + self.AluminumTab
        self.ElectrolyteLiPF6 = 0.15 * cell.ElectrolyteMassOut  # lithium hexafluoro phosphate
        self.ElectrolyteEC = 0.425 * cell.ElectrolyteMassOut  # ethylene carbonate
        self.ElectrolyteDMC = 0.425 * cell.ElectrolyteMassOut  # dimethyl carbonate
        self.PlasticPE = cell.SeparatorMassOut   # polyethylene
        if battery_type == 1:
            self.PlasticPP = [0.056 * cell.ContainerMassOut, 0]   # polypropylene
            self.PlasticPET = [0.127 * cell.ContainerMassOut, 0]  # polyethylene terephthalate
            self.AluminumContainer = [0.817 * cell.ContainerMassOut, 0]
            self.Aluminum = self.Aluminum + self.AluminumContainer
        if cell.ib.AnodeSolvent[1] == 0:
            self.Water = cell.AnodeSolventMassOut
            self.NMPAnode = [0, 0]
            self.NMPCathode = cell.CathodeSolventMassOut
            self.NMP = self.NMPAnode + self.NMPCathode
        else:
            self.Water = [0, 0]
            self.NMPAnode = cell.AnodeSolventMassOut
            self.NMPCathode = cell.CathodeSolventMassOut
            self.NMP = self.NMPAnode + self.NMPCathode
