import numpy as np



[docs]class ProcessPara:

    def __init__(self):
        # mixer details
        self.mixer_diameter = 0.5    # EIRICH mixer type RV11VAC 600 kg 375 l
        self.mixer_power_factor = 7   #
        self.mixer_speed_rps = np.array([17.7, 8.8]) / (self.mixer_diameter * np.pi)  # Bockholt et al 2013
        # mixing_time = np.array([4, 10]) * 60  # Bockholt et al 2013  dry mixing, wet mixing
        self.mixing_time = np.array([2, 18]) * 60  # Eirich ipbc19, 1-2 min and 9-18 min
        # dryer details
        self.anode_dryer_temperature = 100
        self.anode_dryer_time = 15 * 60
        self.cathode_dryer_temperature = 100
        self.cathode_dryer_time = 65 * 60
        # dry room details
        self.air_flow_rate = .008 * 50
        self.air_temperature = 8.0
        self.air_humidity = 75.0
        self.time_dry_room = 35 * 60
        self.desiccant_temperature = -60
        # coating
        self.time_coating = 30  # s
        # calendering
        self.time_calendering = 30  # s
        # slitting
        self.time_slitting = 30  # s
        # stacking
        self.time_stacking = 60  # s
        #  electrolyte filling
        self.time_filling = 40  # s
        # forming and grading
        self.time_formation = 20 * 3600  # s
        self.formation_th = 20
        # total time
        self.total_time = sum(self.mixing_time) + self.anode_dryer_time + self.cathode_dryer_time + self.time_dry_room \
                          + self.time_coating + self.time_calendering + self.time_slitting + self.time_stacking + \
                          self.time_filling + self.time_formation
        self.batch_time = sum(self.mixing_time) + self.time_dry_room + self.time_coating + self.time_calendering + \
                          self.time_slitting + 2 * self.anode_dryer_time
        self.batch_time_hrs = self.batch_time / 3600.
